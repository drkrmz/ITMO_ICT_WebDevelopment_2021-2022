Контейнеризация

Были созданы Docker файлы для запуска бэкенда и фронтенда:

FROM python:3

RUN apt-get update && apt-get upgrade -y && apt-get autoremove && apt-get autoclean

RUN mkdir /library
COPY . /library
WORKDIR /library

RUN pip install --upgrade pip 
RUN pip install django
RUN pip install djangorestframework
RUN pip install django-cors-headers
RUN pip install djoser
RUN pip install djangorestframework-filters

ENTRYPOINT ["python", "manage.py"]
CMD ["runserver", "0.0.0.0:8000"]

FROM node:lts-alpine

RUN mkdir /app
COPY . /app
COPY package*.json /app
WORKDIR /app


RUN npm install
RUN npm install @vue/cli@3.7.0 -g

CMD ["npm", "run", "serve"]

Был создан файл docker-compose для совместного запуска фронтенда и бэкенда:

services:
  django:
    build: ./library_project
    ports: 
      - 8080:8000

  postgres:
    image: postgres:13.3
    environment:
      POSTGRES_DB: "postgres"
      POSTGRES_USER: "postgres"
      POSTGRES_PASSWORD: "postgres"
    volumes:
      - ./dbs/postgres-data:/var/lib/postgresql
    ports:
      - "5432:5432"

  vue:
    build: ./project
    depends_on:
      - django
    ports:
      - 8082:8080

Для запуска необходимо выполнить следующий команды:

docker-compose build
docker-compose up
