2.1.1 Ручные фильтры

Фильтр, который принимает параметр из url-адреса и выводит отфильтрованные данные

views.py

class BookYearList(generics.ListAPIView):
    serializer_class = BookSerializer

    def get_queryset(self):
        year = self.request.query_params.get('year')
        return Book.objects.filter(year=year)

urls.py

re_path('^books/filter/$', BookYearList.as_view())

Фильтр, который принимает 2 параметра из url-адреса и выводит отфильтрованные данные

views.py

class BookNameYear(generics.ListAPIView):
    serializer_class = BookSerializer

    def get_queryset(self):
        queryset = Book.objects.all()
        name = self.request.query_params.get('name')
        year = self.request.query_params.get('year')
        queryset = queryset.filter(year=year)
        queryset = queryset.filter(name=name)
        return queryset

urls.py

re_path('^books/filters/$', BookNameYear.as_view())

Результат:

Фильтр, принимает 2 параметра из url-адреса и выводит отфильтрованные данные, если пользователь авторизован и неотфильтрованные, если не авторизован

views.py

class BookNameYearUser(generics.ListAPIView):
    serializer_class = BookSerializer

    def get_queryset(self):
        queryset = Book.objects.all()
        name = self.request.query_params.get('name')
        year = self.request.query_params.get('year')
        if self.request.user.is_authenticated:
            queryset = queryset.filter(year=year)
            queryset = queryset.filter(name=name)
        return queryset

urls.py

re_path('^books/filters/user/$', BookNameYearUser.as_view())
