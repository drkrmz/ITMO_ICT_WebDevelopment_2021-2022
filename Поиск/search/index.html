    <v-list-item-content>
      <v-text-field v-model="searchTerm" placeholder="Name of the book" @input="searchBooks(page, searchTerm)"></v-text-field>
    </v-list-item-content>

scripts

searchBooks (pageNumber, e) {
      if (sessionStorage.getItem('auth_token')) {
        this.findUser()
      }
      this.search = e
      $.ajax({
        url: 'http://127.0.0.1:8000/library/books/list/?search=' + this.search + '&page=' + pageNumber + '&author=' + this.checked + '&min_year=' + this.range[0] + '&max_year=' + this.range[1],
        type: 'GET',
        success: (response) => {
          this.books = response.results
          this.total = response.count
          this.createRange()
        },
        error: (response) => {
          alert('Something went wrong, please, try again')
        }
      })
    }
